00204 struct S_AlphaProperty
00205 {
00206     /*
00207         In NiAlphaProperty, the flags have the following meaning:
00208 
00209         Bit 0 : alpha blending enable
00210         Bits 1-4 : source blend mode
00211         Bits 5-8 : destination blend mode
00212         Bit 9 : alpha test enable
00213         Bit 10-12 : alpha test mode
00214         Bit 13 : no sorter flag ( disables triangle sorting )
00215 
00216         blend modes (glBlendFunc):
00217         0000 GL_ONE
00218         0001 GL_ZERO
00219         0010 GL_SRC_COLOR
00220         0011 GL_ONE_MINUS_SRC_COLOR
00221         0100 GL_DST_COLOR
00222         0101 GL_ONE_MINUS_DST_COLOR
00223         0110 GL_SRC_ALPHA
00224         0111 GL_ONE_MINUS_SRC_ALPHA
00225         1000 GL_DST_ALPHA
00226         1001 GL_ONE_MINUS_DST_ALPHA
00227         1010 GL_SRC_ALPHA_SATURATE
00228 
00229         test modes (glAlphaFunc):
00230         000 GL_ALWAYS
00231         001 GL_LESS
00232         010 GL_EQUAL
00233         011 GL_LEQUAL
00234         100 GL_GREATER
00235         101 GL_NOTEQUAL
00236         110 GL_GEQUAL
00237         111 GL_NEVER
00238 
00239         Taken from:
00240         http://niftools.sourceforge.net/doc/nif/NiAlphaProperty.html
00241 
00242         Right now we only use standard alpha blending (see the Ogre code
00243         that sets it up) and it appears that this is the only blending
00244         used in the original game. Bloodmoon (along with several mods) do
00245         however use other settings, such as discarding pixel values with
00246         alpha < 1.0. This is faster because we don't have to mess with the
00247         depth stuff like we did for blending. And OGRE has settings for
00248         this too.
00249     */
00250 
00251     // Tested against when certain flags are set (see above.)
00252     unsigned char threshold;

00256         threshold = nif->getChar();
00257     }
